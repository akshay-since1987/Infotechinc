Functions as Classes

function Student (firstName, middleName, lastName) { var middleName = middleName; this.firstName = firstName; this.lastName = lastName; this.getFullName = function () { return this.firstName + " " + middleName + " " + this.lastName; } this.greetsWelcome = function () { return this.firstName + " says \"Hello\""; } } var akshay = new Student("Akshay", "Rajendra", "Khandelwal"); akshay.getFullName();
